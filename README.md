│   ├── src/
│   │   ├── main/
│   │   │   ├── java/com/example/cvbuilder/
│   │   │   │   ├── SplashActivity.java
│   │   │   │   ├── MainActivity.java
│   │   │   │   ├── PersonalDetailsActivity.java
│   │   │   │   └── ... (other activities)
│   │   │   ├── res/
│   │   │   │   ├── layout/
│   │   │   │   │   ├── activity_splash.xml
│   │   │   │   │   ├── activity_main.xml
│   │   │   │   │   └── ... (other layouts)
│   │   │   │   ├── anim/
│   │   │   │   │   └── fade_in.xml
│   │   │   │   └── ... (other resources)
│   │   │   └── AndroidManifest.xml
